import psycopg2
from psycopg2 import OperationalError, sql

def create_connection(db_name, db_user, db_password, db_host, db_port):
    connection = None
    try:
        connection = psycopg2.connect(
            database = db_name,
            user = db_user,
            password = db_password,
            host = db_host,
            port = db_port,
        )
        print("Connection to PostgreSQL DB successful")
    except OperationalError as e:
        print(f"The error '{e}' occurred")
    return connection

def update_vaccine_inventory(connection, location_id, vaccine_id, quantity_change):
    with connection:
        with connection.cursor() as cursor:
            # Set the lock on the specific row
            cursor.execute(
                sql.SQL("SELECT * FROM vaccine_inventory WHERE location_id = %s AND vaccine_id = %s FOR UPDATE"),
                (location_id, vaccine_id),
            )
            # Update the inventory
            cursor.execute(
                sql.SQL("UPDATE vaccine_inventory SET current_inventory = current_inventory + %s WHERE location_id = %s AND vaccine_id = %s"),
                (quantity_change, location_id, vaccine_id),
            )
            print("Update success!")

# Example usage
connection = create_connection("postgres", "postgres", "", "localhost", "")
update_vaccine_inventory(connection, '1', '3', -60)  # Example to increase vaccine inventory by 50 at location 1 for vaccine 3
